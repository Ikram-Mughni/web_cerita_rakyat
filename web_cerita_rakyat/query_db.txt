-- Pastikan menggunakan database yang benar
USE db_cerita_rakyat;

-- 1. Membuat Tabel USERS (Sudah OK)
CREATE TABLE users (
    id INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role ENUM('admin', 'user') NOT NULL DEFAULT 'user'
);

-- 2. Membuat Tabel CERITA (Sudah OK)
CREATE TABLE cerita (
    id INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    image VARCHAR(255) DEFAULT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP()
);

-- 3. Membuat Tabel COMMENTS (Sudah OK dengan perbaikan: merujuk ke 'cerita')
CREATE TABLE comments (
    id INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    cerita_id INT(11) NOT NULL, -- Merujuk ke tabel 'cerita'
    user_id INT(11) NOT NULL,
    comment TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP(),
    FOREIGN KEY (cerita_id) REFERENCES cerita(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
